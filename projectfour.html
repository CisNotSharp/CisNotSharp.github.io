int WINAPI WinMain(HINSTANCE hInst, HINSTANCE
	hPrevInst, LPSTR args, int ncmdshow) {

	// WNDCLASSW is a struct so asign is 0
	WNDCLASSW wc = { 0 };
	// window creation
	wc.hbrBackground = (HBRUSH)COLOR_WINDOW;
	wc.hCursor = (HCURSOR)LoadCursor(NULL, IDC_ARROW);
	wc.hInstance = hInst;
	wc.lpszClassName = L"WindowClass";
	wc.lpszMenuName = L"Main Window";
	wc.lpfnWndProc = WindowProcedure;

	// if window wasnt created
	if (!RegisterClassW(&wc)) {
		return -1;
	}

	CreateWindowW(L"WindowClass", L"Main Window",
		WS_OVERLAPPEDWINDOW | WS_VISIBLE,
		100, 100, 500, 500,
		NULL, NULL, NULL, NULL);

	// struct asign 0
	MSG msg = { 0 };

	while (GetMessage(&msg, NULL, NULL, NULL)) {
		TranslateMessage(&msg);
		DispatchMessageW(&msg);
	}

	return 0;
}

LRESULT CALLBACK WindowProcedure(HWND hWnd, UINT msg, WPARAM wp, LPARAM lp) {
	switch (msg) {
	case WM_COMMAND:

		switch (wp) {
		case 1:
			MessageBeep(MB_OK);
			MessageBox(hWnd, L"Help clicked", L"Help", MB_HELP);
			break;
		case 2:
			MessageBeep(MB_OK);
			MessageBox(hWnd, L"About clicked", L"About", MB_OK);
			break;
		}

		break;
	case WM_CREATE:
		AddMenus(hWnd);
		break;
	case WM_DESTROY:
		PostQuitMessage(0);
		break;
	default :
		return DefWindowProcW(hWnd, msg, wp, lp);
	}
}

void AddMenus(HWND hWnd) {

	hmenu = CreateMenu();
	HMENU hFileMenu = CreateMenu();

	// drop down from file
	AppendMenu(hFileMenu, MF_STRING, FILE_MENU_NEW, L"New");
	AppendMenu(hFileMenu, MF_STRING, FILE_MENU_NEW, L"Open");
	AppendMenu(hFileMenu, MF_STRING, FILE_MENU_NEW, L"Save");
	AppendMenu(hFileMenu, MF_STRING, FILE_MENU_NEW, L"Exit");

	// Main Menu
	AppendMenu(hmenu, MF_POPUP, (UINT_PTR)hFileMenu, L"File");

	// Helpers in Main
	AppendMenu(hmenu, MF_STRING, FILE_MENU_HELP, L"Help");
	AppendMenu(hmenu, MF_STRING, 3, L"About");

	SetMenu(hWnd, hmenu);

}
